282 # If A is grouped, so that each unique element is in a contiguous group, then we only
283 # need to keep track of one element at a time. We replace the elements of A with the
284 # unique elements that we see in the order that we see them. Once we have iterated
285 # through A, we resize A based on the number of unique elements that we see.
286 function _groupedunique!(A::AbstractVector)
287 isempty(A) && return A
288 idxs = eachindex(A)
289 y = first(A)
290 # We always keep the first element

292 count = 1
293 for x in Iterators.drop(A, 1)
294 if !isequal(x, y)
295 y = A[it[1]] = x
296 count += 1
297 it = iterate(idxs, it[2])
298 end
299 end
