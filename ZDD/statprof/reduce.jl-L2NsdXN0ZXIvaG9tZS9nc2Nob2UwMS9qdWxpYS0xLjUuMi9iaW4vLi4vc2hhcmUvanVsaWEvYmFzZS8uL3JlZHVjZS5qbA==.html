53 function _foldl_impl(op::OP, init, itr) where {OP}
54 # Unroll the while loop once; if init is known, the call to op may
55 # be evaluated at compile time
56 y = iterate(itr)
57 y === nothing && return init
58 v = op(init, y[1])
59 while true
60 y = iterate(itr, y[2])
61 y === nothing && break
62 v = op(v, y[1])
63 end
64 return v
65 end

122 """
123 _xfadjoint(op, itr) -> op′, itr′
124
125 Given a pair of reducing function `op` and an iterator `itr`, return a pair
126 `(op′, itr′)` of similar types. If the iterator `itr` is transformed by an
127 iterator transform `ixf` whose adjoint transducer `xf` is known, `op′ = xf(op)`
128 and `itr′ = ixf⁻¹(itr)` is returned. Otherwise, `op` and `itr` are returned
129 as-is. For example, transducer `rf -> MappingRF(f, rf)` is the adjoint of
130 iterator transform `itr -> Generator(f, itr)`.
131
132 Nested iterator transforms are converted recursively. That is to say,
133 given `op` and
134
135 itr = (ixf₁ ∘ ixf₂ ∘ ... ∘ ixfₙ)(itr′)
136
137 what is returned is `itr′` and
138
139 op′ = (xfₙ ∘ ... ∘ xf₂ ∘ xf₁)(op)
140 """

226 # `mapreduce_impl()` is called by `mapreduce()` (via `_mapreduce()`, when `A`
227 # supports linear indexing) and does actual calculations (for `A[ifirst:ilast]` subset).
228 # For efficiency, no parameter validity checks are done, it's the caller's responsibility.
229 # `ifirst:ilast` range is assumed to be a valid non-empty subset of `A` indices.
230
231 # This is a generic implementation of `mapreduce_impl()`,
232 # certain `op` (e.g. `min` and `max`) may have their own specialized versions.
233 @noinline function mapreduce_impl(f, op, A::AbstractArrayOrBroadcasted,
234 ifirst::Integer, ilast::Integer, blksize::Int)
235 if ifirst == ilast
236 @inbounds a1 = A[ifirst]
237 return mapreduce_first(f, op, a1)
238 elseif ifirst + blksize > ilast
239 # sequential portion
240 @inbounds a1 = A[ifirst]
241 @inbounds a2 = A[ifirst+1]
242 v = op(f(a1), f(a2))
243 @simd for i = ifirst + 2 : ilast
244 @inbounds ai = A[i]
245 v = op(v, f(ai))
246 end
247 return v
248 else
249 # pairwise portion
250 imid = (ifirst + ilast) >> 1
251 v1 = mapreduce_impl(f, op, A, ifirst, imid, blksize)
252 v2 = mapreduce_impl(f, op, A, imid+1, ilast, blksize)
253 return op(v1, v2)
254 end
255 end

260 """
261 mapreduce(f, op, itrs...; [init])
262
263 Apply function `f` to each element(s) in `itrs`, and then reduce the result using the binary
264 function `op`. If provided, `init` must be a neutral element for `op` that will be returned
265 for empty collections. It is unspecified whether `init` is used for non-empty collections.
266 In general, it will be necessary to provide `init` to work with empty collections.
267
268 [`mapreduce`](@ref) is functionally equivalent to calling
269 `reduce(op, map(f, itr); init=init)`, but will in general execute faster since no
270 intermediate collection needs to be created. See documentation for [`reduce`](@ref) and
271 [`map`](@ref).
272
273 !!! compat "Julia 1.2"
274 `mapreduce` with multiple iterators requires Julia 1.2 or later.
275
276 # Examples
277 ```jldoctest
278 julia> mapreduce(x->x^2, +, [1:3;]) # == 1 + 4 + 9
279 14
280 ```
281
282 The associativity of the reduction is implementation-dependent. Additionally, some
283 implementations may reuse the return value of `f` for elements that appear multiple times in
284 `itr`. Use [`mapfoldl`](@ref) or [`mapfoldr`](@ref) instead for
285 guaranteed left or right associativity and invocation of `f` for every value.
286 """

423 """
424 reduce(op, itr; [init])
425
426 Reduce the given collection `itr` with the given binary operator `op`. If provided, the
427 initial value `init` must be a neutral element for `op` that will be returned for empty
428 collections. It is unspecified whether `init` is used for non-empty collections.
429
430 For empty collections, providing `init` will be necessary, except for some special cases
431 (e.g. when `op` is one of `+`, `*`, `max`, `min`, `&`, `|`) when Julia can determine the
432 neutral element of `op`.
433
434 Reductions for certain commonly-used operators may have special implementations, and
435 should be used instead: `maximum(itr)`, `minimum(itr)`, `sum(itr)`, `prod(itr)`,
436 `any(itr)`, `all(itr)`.
437
438 The associativity of the reduction is implementation dependent. This means that you can't
439 use non-associative operations like `-` because it is undefined whether `reduce(-,[1,2,3])`
440 should be evaluated as `(1-2)-3` or `1-(2-3)`. Use [`foldl`](@ref) or
441 [`foldr`](@ref) instead for guaranteed left or right associativity.
442
443 Some operations accumulate error. Parallelism will be easier if the reduction can be
444 executed in groups. Future versions of Julia might change the algorithm. Note that the
445 elements are not reordered if you use an ordered collection.
446
447 # Examples
448 ```jldoctest
449 julia> reduce(*, [2; 3; 4])
450 24
451
452 julia> reduce(*, [2; 3; 4]; init=-1)
453 -24
454 ```
455 """
456 reduce(op, itr; kw...) = mapreduce(identity, op, itr; kw...)

570 function mapreduce_impl(f, op::Union{typeof(max), typeof(min)},
571 A::AbstractArrayOrBroadcasted, first::Int, last::Int)
572 a1 = @inbounds A[first]
573 v1 = mapreduce_first(f, op, a1)
574 v2 = v3 = v4 = v1
575 chunk_len = 256
576 start = first + 1
577 simdstop = start + chunk_len - 4
578 while simdstop <= last - 3
579 # short circuit in case of NaN
580 v1 == v1 || return v1
581 v2 == v2 || return v2
582 v3 == v3 || return v3
583 v4 == v4 || return v4
584 @inbounds for i in start:4:simdstop
585 v1 = _fast(op, v1, f(A[i+0]))
586 v2 = _fast(op, v2, f(A[i+1]))
587 v3 = _fast(op, v3, f(A[i+2]))
588 v4 = _fast(op, v4, f(A[i+3]))
589 end
590 checkbounds(A, simdstop+3)
591 start += chunk_len
592 simdstop += chunk_len
593 end
594 v = op(op(v1,v2),op(v3,v4))
595 for i in start:last
596 @inbounds ai = A[i]
597 v = op(v, f(ai))
598 end
599
600 # enforce correct order of 0.0 and -0.0
601 # e.g. maximum([0.0, -0.0]) === 0.0
602 # should hold
603 if isbadzero(op, v)
604 for i in first:last
605 x = @inbounds A[i]
606 isgoodzero(op,x) && return x
607 end
608 end
609 return v
610 end

843 _bool(f) = x->f(x)::Bool

845 """
846 count(p, itr) -> Integer
847 count(itr) -> Integer
848
849 Count the number of elements in `itr` for which predicate `p` returns `true`.
850 If `p` is omitted, counts the number of `true` elements in `itr` (which
851 should be a collection of boolean values).
852
853 # Examples
854 ```jldoctest
855 julia> count(i->(4<=i<=6), [2,3,4,5,6])
856 3
857
858 julia> count([true, false, true, true])
859 3
860 ```
861 """
862 count(itr) = count(identity, itr)
863
864 count(f, itr) = _simple_count(f, itr)
865
866 function _simple_count(pred, itr)
867 n = 0
868 for x in itr
869 n += pred(x)::Bool
870 end
871 return n
872 end
873
874 function count(::typeof(identity), x::Array{Bool})
875 n = 0
876 chunks = length(x) ÷ sizeof(UInt)
877 mask = 0x0101010101010101 % UInt
878 GC.@preserve x begin
879 ptr = Ptr{UInt}(pointer(x))
880 for i in 1:chunks
881 n += count_ones(unsafe_load(ptr, i) & mask)
882 end
883 end
884 for i in sizeof(UInt)*chunks+1:length(x)
885 n += x[i]
886 end
887 return n
888 end
