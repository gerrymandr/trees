191 """
192 SimpleDiGraph(edge_list::Vector)
193
194 Construct a `SimpleDiGraph` from a vector of edges.
195 The element type is taken from the edges in `edge_list`.
196 The number of vertices is the highest that is used in an edge in `edge_list`.
197
198 ### Implementation Notes
199 This constructor works the fastest when `edge_list` is sorted
200 by the lexical ordering and does not contain any duplicates.
201
202 ### See also
203 [`SimpleDiGraphFromIterator`](@ref)
204
205 ## Examples
206 ```jldoctest
207
208 julia> el = Edge.([ (1, 3), (1, 5), (3, 1) ])
209 julia> SimpleDiGraph(el)
210 {5, 3} directed simple Int64 graph
211 ```

213 function SimpleDiGraph(edge_list::Vector{SimpleDiGraphEdge{T}}) where T <: Integer
214 nvg = zero(T)
215 @inbounds(
216 for e in edge_list
217 nvg = max(nvg, src(e), dst(e))
218 end)
219
220 list_sizes_out = ones(Int, nvg)
221 list_sizes_in = ones(Int, nvg)
222 degs_out = zeros(Int, nvg)
223 degs_in = zeros(Int, nvg)
224 @inbounds(
225 for e in edge_list
226 s, d = src(e), dst(e)
227 (s >= 1 && d >= 1) || continue
228 degs_out[s] += 1
229 degs_in[d] += 1
230 end)
231
232 fadjlist = Vector{Vector{T}}(undef, nvg)
233 badjlist = Vector{Vector{T}}(undef, nvg)
234 @inbounds(
235 for v in 1:nvg
236 fadjlist[v] = Vector{T}(undef, degs_out[v])
237 badjlist[v] = Vector{T}(undef, degs_in[v])
238 end)
239
240 @inbounds(
241 for e in edge_list
242 s, d = src(e), dst(e)
243 (s >= 1 && d >= 1) || continue
244 fadjlist[s][list_sizes_out[s]] = d
245 list_sizes_out[s] += 1
246 badjlist[d][list_sizes_in[d]] = s
247 list_sizes_in[d] += 1
248 end)
249
250 neg = cleanupedges!(fadjlist, badjlist)
251 g = SimpleDiGraph{T}()
252 g.fadjlist = fadjlist
253 g.badjlist = badjlist
254 g.ne = neg
255
256 return g
257 end
